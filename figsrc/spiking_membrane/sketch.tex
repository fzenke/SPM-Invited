\begin{tikzpicture}

	% Place nodes 
	\foreach \t in {-2,...,1} {

	}
	\node [circle, minimum height=2.0em, fill=outputcolor] (out) at (0,2) {};


    % Draw edges
	\foreach \t in {-2,...,1} {

	}

	\node [font=\sffamily] at (0,-0.6) {Input neurons};
	\node [font=\sffamily] at (0,2.6) {Output neuron};
